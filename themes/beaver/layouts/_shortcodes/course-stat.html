{{- /* Course stats from the single source data/course_sequence.yaml. Usage in body prose (positional param): course-stat chapters -> count of sequence entries excluding the Overview course-stat modules -> count of distinct module numbers in the sequence any other key -> looked up in the yaml's `stats:` map (e.g. artifacts) Frontmatter and cover binaries cannot run shortcodes - use near numbers ("20+ chapters") there instead. */ -}} {{- $key := .Get 0 -}} {{- $data := site.Data.course_sequence -}} {{- if eq $key "chapters" -}} {{- $n := 0 -}} {{- range $data.sequence -}} {{- if ne .module "Overview" -}}{{- $n = add $n 1 -}}{{- end -}} {{- end -}} {{- $n -}} {{- else if eq $key "modules" -}} {{- $mods := slice -}} {{- range $data.sequence -}} {{- $m := index (findRE `^Lesson (\d+)` .module 1) 0 -}} {{- if and $m (not (in $mods $m)) -}}{{- $mods = $mods | append $m -}}{{- end -}} {{- end -}} {{- len $mods -}} {{- else -}} {{- $v := index (default dict $data.stats) $key -}} {{- if eq $v nil -}} {{- errorf "course-stat: unknown key %q (page %s) - add it under stats: in data/course_sequence.yaml" $key page.Path -}} {{- end -}} {{- $v -}} {{- end -}}
